// Lock is a mutually exclusive lock that supports canceling acquire operation.
type Lock chan struct{}

// NewLock returns a new Lock that allows at most one goroutine to acquire it.

	return make(Lock, 1)
}

	case <-.Done():
		return .Err()
	case  <- struct{}{}:
		return nil
	}
}
