func ( *UUID) ( []byte) {
	// 13th character is "4"
	[6] = ([6] & 0x0f) | 0x40 // Version 4
	// 17th character is "8", "9", "a", or "b"
	[8] = ([8] & 0x3f) | 0x80 // Variant most significant bits are 10x where x can be either 1 or 0
}
